function	script	F_OpenExpressStorage	{
	mes callfunc("P_Purple", "[Armazém Expresso]");
	mes "Você tem certeza que deseja consumir este Armazém Expresso?";
	mes "\n";
	mes callfunc("P_Orange", "Esta é uma ação irreversível e não estornável.");
	switch(prompt("Sim", "Não")) {
		case 1:
			openstorage;
			close;
			break;
		case 2:
			getitem 33335, 1;
			break;
		case 255:
			getitem 33335, 1;
			end;
			break;
	}

	close;
}
